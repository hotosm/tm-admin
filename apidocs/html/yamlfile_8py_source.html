52 def getEntries(self):
53 """
54 Convert the list from the YAML file into a searcable data structure
55 """
56 table = list(self.yaml[0].keys())[0]
57 for entry in self.yaml[0][table]:
58 for key, values in entry.items():
59 self.data[key] = {'array': False, 'required': False, 'share': False}
60 for item in values:
61 # The only values that's a string is the data type
62 if type(item) == str:
63 if item[:7] == 'public.':
64 self.data[key]['datatype'] = item[7:].capitalize()
65 else:
66 self.data[key]['datatype'] = item
67 else:
68 [[k, v]] = item.items()
69 self.data[key][k] = v
70
71 return self.data
