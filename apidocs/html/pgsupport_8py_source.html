132 """
133 Insert a record in a database table. All the primary tables auto-increment
134 the id column. If id is set in the record, then it uses that value, otherwise
135 it increments.
136
137 Args:
138 records (dict): The record data
139
140 Returns:
141 (bool): Whether the record got inserted into the database
142 """
143 # log.warning(f"--- insertRecords(): ---")
144 if not self.table:
145 log.error(f"Not connected to the database!")
146 return False
147
148 data = dict()
149 for entry in records:
150 for key, value in entry.data.items():
151 val = self.types[key]
152 if not value:
153 continue
154 if type(value) == list:
155 fo = str(value)
156 data[key] = f"XARRAY{str(value)}X"
157 elif type(value) == dict:
158 # It's for a jsonb column
159 # a dict uses single quotes, postgres wants double quotes.
160 newval = str(value).replace("'", '"')
161 data[key] = f"{newval}"
162 continue
163 # This is a bit ugly. When passing a geometry we need to get rid of the
164 # double quote around this value.
165 elif val == "Polygon":
166 data[key] = f"XST_GeomFromText('{value}')X"
167 elif val == "Point":
168 data[key] = f"XST_GeomFromText('{value}')X"
169 elif val[:7] == "public.":
170 # FIXME: unfortunately, the supposed enums are a mix of
171 # the string value or the integer value. Ideally this should
172 # all get cleaned up, but till then handle both.
173 # print(f"FIXME: {key}: {type(value)}")
174 tmtype = val[7:].capitalize()
175 if type(value) == str:
176 data[key] = value
177 elif type(value) == int:
178 # All we have is the value, so instantiate the Enum
179 # to get the name. Also check if it's an array.
180 if tmtype[-2:] == "[]":
181 obj = eval("tm_admin.types_tm.%s(%s)" % (tmtype[:-2], value))
182 data[key] = "{%s}" % obj.name
183 else:
184 obj = eval(f"tm_admin.types_tm.{tmtype}({value})")
185 data[key] = obj.name
186 else:
187 data[key] = value
188
189 keys = str(list(data.keys())).replace("'", "")[1:-1]
190 # This is a bit ugly. When passing a geometry we need to get rid of the
191 # double quote around this value.
192 values = str(list(data.values()))[1:-1]
193 # FIXME: dealing with quoting between python and postgres is a hassle
194 foo = values.replace('"X', "").replace('X"', "")
195 # print(f"1: {foo}")
196 foo = foo.replace("'X", "").replace("X'", "")
197 # print(f"2: {foo}")
198 sql = f"INSERT INTO {self.table}({keys}) VALUES({foo}) RETURNING id"
199 # print(sql)
200 result = await self.execute(sql)
201 # print(result)
202 if len(result) > 0:
203 return result[0]['id']
204 else:
205 return list()

300 """
301 Get columns from a database table.
302
303 Args:
304 column (list): The column names to include in the output
305 where (list): The conditions to limit the records, might return
306 more than one
307 jsonb (bool): Whether to loop through and convert the jsonb to a dict
308
309 Returns:
310 (list): The data for this column
311 """
312 get = str(columns)[1:-1].replace("'", "")
313
314 check = str()
315 if where:
316 for k, v in where.items():
317 if str(type(v))[:5] == "<enum":
318 check = f" {k} = '{v.name}'"
319 elif type(v) == dict:
320 # It's a query including a jsonb column
321 for k1, v1 in v.items():
322 # teams->'teams' @? '$[*] ? (@.role == 1)
323 # FIXME: is it an internal Enum ?
324 check = f"{k}->'{k}' @? '$[*] ? (@.{k1} == {v1})'"
325 continue
326 elif v == 'null':
327 check = f"{k} IS NOT NULL"
328 elif len(check) == 0:
329 check = f"{k}={v}"
330 sql = f"SELECT {get} FROM {self.table} WHERE {check}"
331 else:
332 sql = f"SELECT {get} FROM {self.table}"
333 # print(sql)
334 results = await self.execute(sql)
335
336 data = list()
337 for record in results:
338 # print(record)
