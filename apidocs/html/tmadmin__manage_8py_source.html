142  def readDiff(self,
143  diff: str,
144  ):
145  """
146  Read in a diff produced by git diff into a data structure
147  that can be used to updgrade a table's schema.
148 
149  Args:
150  diff (str): the name of the SQL diff file
151 
152  Returns:
153  (dict): The columns to drop or add from a table
154  """
155  with open(diff, 'r') as file:
156  for line in file.readlines():
157  # it's the header
158  if line[:3] == '---' or line[:3] == '+++' or line[:2] == '@@':
159  continue
160  # Ignore code block
161  if line[0] == ' ':
162  continue
163  tmp = ' '.join(line.split()).split()
164  if tmp[0] == '-':
165  self.columnscolumns['drop'].append(tmp[1])
166  if tmp[0] == '+':
167  if tmp[1] in self.columnscolumns['drop']:
168  self.columnscolumns['drop'].remove(tmp[1])
169  continue
170  if len(tmp) == 3:
171  self.columnscolumns['add'].append({tmp[1]: tmp[2][:-1]})
172  elif len(tmp) == 4:
173  self.columnscolumns['add'].append({tmp[1]: f"{tmp[2]} {tmp[3][:-1]}"})
174  return self.columnscolumns
