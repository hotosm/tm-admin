295  """
296  Create the source for an SQL table.
297 
298  Returns:
299  (str): The protobuf message source.
300  """
301  out = "-- Do not edit this file, it's generated from the yaml file\n\n"
302  sequence = list()
303  for entry in self.yamlyaml.yaml:
304  [[table, values]] = entry.items()
305  out += f"DROP TABLE IF EXISTS public.{table} CASCADE;\n"
306  out += f"CREATE TABLE public.{table} (\n"
307  unique = list()
308  typedef = ""
309  for line in values:
310  # these are usually from the types.yaml file, which have no
311  # settings beyond the enum value.
312  if type(line) == str:
313  # print(f"SQL TABLE: {typedef} {line}")
314  typedef = table
315  continue
316  [[k, v]] = line.items()
317  required = ""
318  array = ""
319  public = False
320  primary = list()
321  for item in v:
322  if type(item) == dict:
323  if 'sequence' in item and item['sequence']:
324  sequence.append(k)
325  if 'required' in item and item['required']:
326  required = ' NOT NULL'
327  if 'array' in item and item['array']:
328  array = "[]"
329  if 'unique' in item and item['unique']:
330  unique.append(k)
331  if 'primary' in item and item['primary']:
332  primary.append(k)
333  if len(v) >= 2:
334  if 'required' in v[1] and v[1]['required']:
335  required = ' NOT NULL'
336  if type(item) == str:
337  if item[:7] == 'public.' and item[15:8] != 'geometry':
338  public = True
339  # elif item[15:8] == 'geometry':
340  # out += f"\t{k} {self.yaml2py[v[0]]}{array}{required},\n"
341  if public:
342  out += f"\t{k} {v[0]}{array}{required},\n"
343  else:
344  # print(v)
345  # FIXME: if this produces an error, check the yaml file as this
346  # usually means the type field isn't first in the list.
347  if v[0][:6] == 'table.':
348  out += f"\t{k} {v[0][6:]}{array},\n"
349  else:
350  out += f"\t{k} {self.yaml2sql[v[0]]}{array}{required},\n"
351  if len(unique) > 0:
352  keys = str(unique).replace("'", "")[1:-1];
353  out += f"\tUNIQUE({keys})\n);\n"
354  if out[-2:] == ',\n':
355  out = f"{out[:-2]}\n);\n\n"
356  if len(sequence) > 0:
357  for key in sequence:
358  out += f"""
359 DROP SEQUENCE IF EXISTS public.{table}_{key}_seq CASCADE;
360 CREATE SEQUENCE public.{table}_{key}_seq
361  START WITH 1
362  INCREMENT BY 1
363  NO MINVALUE
364  NO MAXVALUE
365  CACHE 1;
366 
367 """
368  if len(primary) > 0:
369  keys = str(primary).replace("'", "")[1:-1];
370  out += f"\tALTER TABLE {table} ADD CONSTRAINT {table}_pkey PRIMARY KEY({keys})\n);\n"
371 
372  return out + "\n"
