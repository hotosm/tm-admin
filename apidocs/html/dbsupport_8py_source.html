79 async def createTable(self,
80 obj,
81 ):
82 """
83 Create a table in a postgres database.
84
85 Args:
86 obj: The config data for the table.
87 """
88 sql = f"INSERT INTO {self.table}(id, "
89 for column,value in obj.data.items():
90 # print(f"{column} is {type(value)}")
91 if type(value) == str:
92 # FIXME: for now ignore timestamps, as they're meaningless
93 # between projects
94 try:
95 if parse(value):
96 continue
97 except:
98 # it's a string, but not a date
99 pass
100 if value is not None:
101 sql += f"{column},"
102 sql = sql[:-1]
103 sql += f") VALUES("
104 for column,value in obj.data.items():
105 try:
106 if parse(value):
107 continue
108 except:
109 pass
110 if column == 'id':
111 sql += f"nextval('public.{self.table}_id_seq'),"
112 continue
113 if value is None:
114 continue
115 elif type(value) == datetime:
116 continue
117 elif type(value) == int:
118 sql += f"{value},"
119 elif type(value) == bool:
120 if value:
121 sql += f"true,"
122 else:
123 sql += f"false,"
124 elif type(value) == str:
125 sql += f"'{value}',"
126
127 #print(sql[:-1])
128 result = await self.pg.execute(f"{sql[:-1]});")
